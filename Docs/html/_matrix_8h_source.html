392  /// SetInverse the matrix.
393  const bool SetInverse()
394  {
395  int i, j, k, swap;
396  T temp[16], t;
397 
398  memcpy(temp, m, 16*sizeof(T));
399  SetIdentity();
400 
401  for (i = 0; i < 4; i++)
402  {
403  swap = i;
404  for (j = i + 1; j < 4; j++)
405  {
406  if (fabs(temp[j*4 + i]) > fabs(temp[i*4 + i]))
407  {
408  swap = j;
409  }
410  }
411 
412  if (swap != i)
413  {
414  for (k = 0; k < 4; k++)
415  {
416  t = temp[i*4 + k];
417  temp[i*4 + k] = temp[swap*4 + k];
418  temp[swap*4 + k] = t;
419 
420  t = m[i*4 + k];
421  m[i*4 + k] = m[swap*4 + k];
422  m[swap*4 + k] = t;
423  }
424  }
425 
426  if(!temp[i*4 + i])
427  return false;
428 
429  t = temp[i*4 + i];
430 
431  for (k = 0; k < 4; k++)
432  {
433  temp[i*4 + k] /= t;
434  m[i*4 + k] = m[i*4 + k] / t;
435  }
436 
437  for (j = 0; j < 4; j++)
438  {
439  if (j != i)
440  {
441  t = temp[j*4 + i];
442  for (k = 0; k < 4; k++)
443  {
444  temp[j*4 + k] -= temp[i*4 + k] * t;
445  m[j*4 + k] = m[j*4 + k] - m[i*4 + k] * t;
446  }
447  }
448  }
449  }
450  return true;
