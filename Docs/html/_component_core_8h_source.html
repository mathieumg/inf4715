110 {
111  std::string classname = VCN::ExtractTypeName<T>();
112  assert(GetInstantiationMap().find(classname) == std::end(GetInstantiationMap()));
113 
114  GetInstantiationMap().insert(
115  InstantiationPair(classname,
116  [](const VCNIComponent::Parameters& params, VCNNode* owner) -> VCNIComponent::Ptr
117  {
118  //All components must have a parameterless constructor
119  VCNIComponent::Ptr value = VCNIComponent::Ptr( ComponentTraits<T>::FactoryFunction() );
120  owner->AddComponent(value);
121  value->Initialise(params);
122  return value;
123  }));
124
