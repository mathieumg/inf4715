65 template<typename EventType>
66 template<typename ListenerType>
67 VCNEvent<EventType>::VCNEvent(ListenerType* listener, void (ListenerType::*callback)(const EventType&))
68  : VCNIEvent( VCN::ExtractTypeName<EventType>(), listener )
69 {
70  static_assert( std::is_base_of<VCNIListener, ListenerType>::value, "listener must be of type VCNIListener");
71  /*
72  This constraint is only to make sure we don't have any pointer problems. This would only be the case if
73  we were playing with multiple inheritance and calling this function with a template parameter for the listener
74  at different levels (lets say VCNIComponent and PhysXComponent), in this case the values of the pointers may be
75  different but will definitely be pointing to the same listener.
76 
77  Hence the behavior could be drastically different. If the values are different then the two are considered different listeners
78  so each caller has to call unregister (even if they are actually the same caller), which is ok as it gives us more control. The problem
79  is that it won't be homogeneous. One call to remove all listeners could remove everything for some, and not others depending on the inheritance
80  hierarchy.
81 
82  That is why we enforce the use of VCNIListener, it could be anything, it just all has to be the same class.
83 
84  Although if VCNIListener is found multiple times in the hierarchy, each element inheriting
85  from VCNIListener would have to unregister (if said object has indeed registered himself).
86  */
87  m_callback = std::bind(callback, listener, std::placeholders::_1);
88 }
